<!-- Javascript -->
<script>
$(function () {
        $(".square").draggable();
        $(".squaredotted").droppable({
                drop: function (event, ui) {
                        $("#info").html("dropped!");
                },
                over: function (event, ui) {
                        $("#info").html("moving in!");
                },
                out: function (event, ui) {
                        $("#info").html("moving out!");
                }
        });
});
</script>
